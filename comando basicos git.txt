git init	#inicial o controle de versao do projeto

git config --global user.name "colocar_o_nome_de_usuario_local"
git config --global user.email "colocar_seu_email"

git status	#verificar se houve alteracao na pasta do projeto

git add . / git add nome_do_arquivo	#para incluir arquivo no proximo commit (proxima versao)
git commit -m "descricao da alteracao feita no arquivo"	#para gerarmos uma versao do arquivo
git push origin main	#subir arquivo para o git hub

git log		#para verificar as versoes do commit
git reset --hard "id_do_commit_que_eu_quero_voltar"	#para voltar até uma versão anterior
